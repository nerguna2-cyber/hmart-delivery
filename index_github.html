'block' : 'none'; } async function specialDepart() { var vehicle = document.getElementById('special-vehicle').value; var depKm = document.getElementById('special-depkm').value; var codrv = document.getElementById('special-codrv') ? document.getElementById('special-codrv').value : ''; var notes = document.getElementById('special-notes').value; var customEl = document.getElementById('special-type-custom'); var runType = selectedSpecialType === '__other__' ? (customEl ? customEl.value.trim() || 'Other' : 'Other') : selectedSpecialType; if(!curDriver) { toast('drv-toast', 'Select your name first', false); return; } if(!runType) { toast('drv-toast', 'Select run type', false); return; } if(!vehicle) { toast('drv-toast', 'Select vehicle', false); return; } if(!depKm) { toast('drv-toast', 'Enter departure KM', false); return; } var vSel = document.getElementById('special-vehicle'); var plate = (vSel.options[vSel.selectedIndex] && vSel.options[vSel.selectedIndex].dataset.plate) || ''; var btn = document.getElementById('special-depart-btn'); btn.textContent = 'Departing...'; btn.disabled = true; try { var r = await apiPost('submitDeparture', { driver: curDriver, vehicle: vehicle, plate: plate, destination: '🚨 ' + runType + (notes ? ' — ' + notes : ''), depKm: parseFloat(depKm), coDriver: codrv, orderIds: [] }); btn.textContent = '🚨 Special / Emergency Depart'; btn.disabled = false; if(r && r.success) { specialRunActive = true; var infoEl = document.getElementById('special-run-info'); if(infoEl) infoEl.innerHTML = '🚨 ' + runType + ' | ' + vehicle + ' | Dep: ' + depKm + ' km'; document.getElementById('special-run-form').style.display = 'none'; document.getElementById('special-active').style.display = 'block'; var toggleBtn = document.getElementById('special-toggle-btn'); if(toggleBtn) toggleBtn.style.display = 'none'; } else toast('drv-toast', 'Error departing', false); } catch(e) { btn.textContent = '🚨 Special / Emergency Depart'; btn.disabled = false; toast('drv-toast', 'Error: ' + e.message, false); } } async function specialArrive() { var arrKm = document.getElementById('special-arrkm').value; var remarks = document.getElementById('special-remarks').value; if(!arrKm) { toast('drv-toast', 'Enter arrival KM', false); return; } var btn = document.querySelector('#special-active .btn-arrive'); btn.textContent = 'Saving...'; btn.disabled = true; try { var r = await apiPost('submitArrival', { driver: curDriver, arrKm: parseFloat(arrKm), remarks: remarks }); btn.textContent = 'Complete Special Run ✓'; btn.disabled = false; if(r && r.success) { toast('drv-toast', 'Special run completed! +' + r.distance + ' km', true); specialRunActive = false; try { localStorage.removeItem('hmart_special_run_' + curDriver); } catch(e2) {} setTimeout(function() { document.getElementById('special-active').style.display = 'none'; document.getElementById('special-run-form').style.display = 'none'; var toggleBtn = document.getElementById('special-toggle-btn'); if(toggleBtn) { toggleBtn.style.display = 'block'; toggleBtn.textContent = '+ Open'; } ['special-depkm','special-arrkm','special-remarks','special-notes'].forEach(function(id) { var el = document.getElementById(id); if(el) el.value = ''; }); var ci = document.getElementById('special-type-custom'); if(ci) { ci.value = ''; ci.style.display = 'none'; } selectedSpecialType = ''; document.querySelectorAll('.special-type-btn').forEach(function(b) { b.style.background = '#fff'; b.style.color = '#374151'; b.style.borderColor = '#D1D5DB'; }); }, 2000); } else toast('drv-toast', (r && r.message) || 'Error', false); } catch(e) { btn.textContent = 'Complete Special Run ✓'; btn.disabled = false; toast('drv-toast', 'Error: ' + e.message, false); } }
